  http://{{ include "loop.fullname" . }}-gateway:{{ .Values.gateway.service.port }}

{{- if .Values.ingress.enabled }}

  https://{{ .Values.ingress.host }}/v1/cp
  https://{{ .Values.ingress.host }}/v1/runtime
  https://{{ .Values.ingress.host }}/v1/llm
{{- else }}

Ingress is disabled. Run `kubectl port-forward svc/{{ include "loop.fullname" . }}-gateway {{ .Values.gateway.service.port }}` to reach the gateway from your laptop.
{{- end }}
